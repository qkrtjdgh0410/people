# app_eda.py 내부에서 사용될 예시 코드

import streamlit as st
import pandas as pd
import numpy as np

def preprocess_data(df):
    # '세종' 지역의 '-'을 0으로 변경
    df.loc[df['지역'] == '세종'] = df.loc[df['지역'] == '세종'].replace('-', 0)

    # 숫자형 컬럼 변환
    cols_to_convert = ['인구', '출생아수(명)', '사망자수(명)']
    for col in cols_to_convert:
        df[col] = pd.to_numeric(df[col], errors='coerce')

    return df

def display_basic_stats(df):
    st.subheader("DataFrame Info")
    buffer = []
    df.info(buf=buffer)
    s = "\n".join(buffer)
    st.text(s)

    st.subheader("Descriptive Statistics")
    st.write(df.describe())

# Streamlit 인터페이스 예시
st.title("Population Trends EDA App")

uploaded_file = st.file_uploader("Upload population_trends.csv", type="csv")

if uploaded_file is not None:
    df = pd.read_csv(uploaded_file)
    df = preprocess_data(df)
    display_basic_stats(df)
